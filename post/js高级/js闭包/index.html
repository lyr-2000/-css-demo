// 多次操作变成一次操作
function debounce(func, ms = 1000) {
  let timer;
  return function (...args) {
    if (timer) { // 重复调用就会清除上一次定时任务
      clearTimeout(timer)
    }
    timer = setTimeout(() => {
      func.apply(this, args)
    }, ms)
  }
}

//让水龙头的流水变小一点，一定时间内调用一次函数
 function throttle(func, delay) {
    var prev = Date.now();
    return function() {
        var context = this;   //this指向window
        var args = arguments;
        var now = Date.now();
        if (now - prev >= delay) {
            func.apply(context, args);
            prev = Date.now();
        }
    }
}

function throttle(func,delay) {
    let timer = null;
    return function() {
        if(!timer) {
            let ctx = this;
            let args = arguments;
            timer = setTimeout(()=> {
                func.apply(ctx,args)
                timer = null
            },delay)
        }
    }
}
